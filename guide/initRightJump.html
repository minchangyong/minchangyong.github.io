# 示例代码

let params: {
	imageUrl: 'https://qiniu-image.qtshe.com/logo.jpg'
}
// 新版本
qts.invoke('saveImageToPhotosAlbum', params).then((res) => {
	console.log(res) // 返回true则是用户已允许开启相册权限
})

// 老版本
if (util.isAndroidApp()) { // 安卓使用网络图片即可
	qts.saveImageToPhotosAlbum('https://qiniu-image.qtshe.com/20210315poster.png').then((res) => {
		console.log(res)
	})
} else if (util.isIosApp()) { // ios需要将网络图片转为base64图片，节省客户端下载图片的时间
	util.convertImgToBase64('https://qiniu-image.qtshe.com/20210315poster.png', (res) => {
		jsBridge.saveImageToPhotosAlbum(res)
	})
}
