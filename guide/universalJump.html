# qts.universalJump()

以 Promise 风格 调用：

支持

# 功能描述

# 跳转客户端任意界面的方法

# 示例代码

// 单参数，示例中的 静态页面 参数及值
let params = {
	targetUrl: 'https://m.qtshe.com'
}

// 多参数，示例中的 分享页面 参数及值
let params = {
    targetUrl: 'https://m.qtshe.com',
    shareContent: '一次兼职、一次成长',
    shareTitle: '青团社兼职',
    logoUrl: 'https://qiniu-image.qtshe.com/logo.png'
}

qts.universalJump('USER_STATIC_PAGE', params).then((res) => {
	console.log(res)
})
